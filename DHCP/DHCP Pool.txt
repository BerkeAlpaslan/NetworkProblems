DHCP Limited IP Pool
# Setup Interface
Router> enable
Router# configure terminal
Router(config)# interface FastEthernet0/0
Router(config-if)# ip address 192.168.1.1 255.255.255.0
Router(config-if)# no shutdown
Router(config-if)# exit

# DHCP Pool
Router(config)# ip dhcp pool LIMITED_POOL
Router(dhcp-config)# network 192.168.1.0 255.255.255.0
Router(dhcp-config)# default-router 192.168.1.1
Router(dhcp-config)# dns-server 8.8.8.8
Router(dhcp-config)# exit

# Only 3 IP (192.168.1.2, 1.3, 1.4)
Router(config)# ip dhcp excluded-address 192.168.1.1 192.168.1.1
Router(config)# ip dhcp excluded-address 192.168.1.5 192.168.1.254
